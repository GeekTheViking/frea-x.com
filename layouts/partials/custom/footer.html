// Hent data fra Hugo genereret JSON fetch('/data/all-vards.json') .then(response => response.json()) .then(data => { // Bland alle vards sammen const allVards = [...data.ekkoer, ...data.artefakter]; // Tilføj til en global variabel (eller gem det i localStorage for hurtigere adgang) window.gtvVards = allVards; // Tilføj event listener nu, når data er klar document.getElementById('dontPushBtn').addEventListener('click', () => { if (!window.gtvVards || window.gtvVards.length === 0) { alert("Ingen skatte fundet i lageret endnu!"); return; } const randomVard = window.gtvVards[Math.floor(Math.random() * window.gtvVards.length)]; document.getElementById('randomQuoteText').innerHTML = `${randomVard.title}:
"${randomVard.content}"`; document.getElementById('easterEggModal').style.display = 'flex'; }); }) .catch(err => console.error("Kunne ikke hente Data-Varder:", err));
